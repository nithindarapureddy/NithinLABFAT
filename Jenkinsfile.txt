pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Build') {
            steps {
                bat 'mvn clean install'
            }
        }

        stage('Docker Build') {
            steps {
                bat 'docker build -t your-docker-image-name .'
            }
        }

        stage('Docker Push') {
            steps {
                bat 'docker push your-docker-image-name'
            }
        }
    }

    post {
        failure {
            script {
                echo 'Pipeline failed! Check the logs for details.'
                currentBuild.result = 'FAILURE'
            }
        }
    }
}
